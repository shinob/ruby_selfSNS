CREATE TABLE `notes`(
  id integer primary key,
  user_id integer,
  note_id integer,
  note_type text,
  post_date date,
  comment text
);

CREATE TABLE `likes` (
  id integer primary key,
  post_date date,
  note_id integer,
  user_id integer
);

CREATE TABLE `users`(
  id integer primary key,
  disp_name text,
  name text,
  password text,
  auth_type text
);

CREATE TABLE `profiles`(
  id integer primary key,
  user_id integer,
  title text,
  value text
);
